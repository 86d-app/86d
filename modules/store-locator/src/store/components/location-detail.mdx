<div className="mx-auto max-w-3xl">
  {props.location?.imageUrl && (
    <div className="mb-6 aspect-video overflow-hidden rounded-lg">
      <img
        src={props.location.imageUrl}
        alt={props.location.name}
        className="h-full w-full object-cover"
      />
    </div>
  )}

  <div className="flex items-start justify-between gap-4">
    <div>
      <h1 className="text-3xl font-bold text-foreground">{props.location?.name}</h1>
      {props.location?.description && (
        <p className="mt-2 text-muted-foreground">{props.location.description}</p>
      )}
    </div>
    {props.isOpen !== undefined && (
      <span className={`mt-1 shrink-0 rounded-full px-3 py-1 text-sm font-medium ${
        props.isOpen
          ? "bg-green-500/10 text-green-700"
          : "bg-red-500/10 text-red-700"
      }`}>
        {props.isOpen ? "Open Now" : "Closed"}
      </span>
    )}
  </div>

  <div className="mt-8 grid gap-8 sm:grid-cols-2">
    <div>
      <h2 className="mb-3 text-lg font-semibold text-foreground">Address</h2>
      <p className="text-muted-foreground">
        {props.location?.address}<br />
        {props.location?.city}
        {props.location?.state ? `, ${props.location.state}` : ""}
        {props.location?.postalCode ? ` ${props.location.postalCode}` : ""}<br />
        {props.location?.country}
      </p>
    </div>

    <div>
      <h2 className="mb-3 text-lg font-semibold text-foreground">Contact</h2>
      <div className="space-y-1 text-muted-foreground">
        {props.location?.phone && (
          <p><a href={`tel:${props.location.phone}`} className="hover:text-primary">{props.location.phone}</a></p>
        )}
        {props.location?.email && (
          <p><a href={`mailto:${props.location.email}`} className="hover:text-primary">{props.location.email}</a></p>
        )}
        {props.location?.website && (
          <p><a href={props.location.website} target="_blank" rel="noopener noreferrer" className="hover:text-primary">{props.location.website}</a></p>
        )}
      </div>
    </div>
  </div>

  {props.location?.hours && Object.keys(props.location.hours).length > 0 && (
    <div className="mt-8">
      <h2 className="mb-3 text-lg font-semibold text-foreground">Hours</h2>
      <div className="space-y-1">
        {["monday", "tuesday", "wednesday", "thursday", "friday", "saturday", "sunday"].map((day) => {
          const dayHours = props.location?.hours?.[day];
          const isToday = props.currentDay?.toLowerCase() === day;
          return (
            <div
              key={day}
              className={`flex justify-between rounded px-3 py-1.5 text-sm ${
                isToday ? "bg-primary/5 font-medium" : ""
              }`}
            >
              <span className="capitalize text-foreground">{day}</span>
              <span className="text-muted-foreground">
                {dayHours?.closed ? "Closed" : dayHours ? `${dayHours.open} – ${dayHours.close}` : "—"}
              </span>
            </div>
          );
        })}
      </div>
    </div>
  )}

  {props.location?.amenities?.length > 0 && (
    <div className="mt-8">
      <h2 className="mb-3 text-lg font-semibold text-foreground">Amenities</h2>
      <div className="flex flex-wrap gap-2">
        {props.location.amenities.map((amenity) => (
          <span
            key={amenity}
            className="rounded-full bg-muted px-3 py-1 text-sm text-muted-foreground capitalize"
          >
            {amenity.replace(/-/g, " ")}
          </span>
        ))}
      </div>
    </div>
  )}

  {props.location?.pickupEnabled && (
    <div className="mt-8 rounded-lg border border-green-200 bg-green-50 p-4">
      <p className="font-medium text-green-800">In-Store Pickup Available</p>
      <p className="mt-1 text-sm text-green-700">
        Order online and pick up at this location.
      </p>
    </div>
  )}
</div>
