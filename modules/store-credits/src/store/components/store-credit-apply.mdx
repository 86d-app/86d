{props.applied ? (
  <div className="rounded-lg border border-green-200 bg-green-50 p-3 dark:border-green-800 dark:bg-green-950">
    <p className="font-medium text-green-700 text-sm dark:text-green-300">
      Store credit applied: {props.formatCurrency(props.applied.amount, props.currency)}
    </p>
    <p className="mt-0.5 text-green-600 text-xs dark:text-green-400">
      Remaining balance: {props.formatCurrency(props.applied.remaining, props.currency)}
    </p>
  </div>
) : props.balanceLoading ? (
  <div className="rounded-lg border border-gray-200 bg-white p-3 dark:border-gray-800 dark:bg-gray-900">
    <p className="text-gray-400 text-sm dark:text-gray-500">
      Checking store credit...
    </p>
  </div>
) : props.balance <= 0 || props.status !== "active" ? (
  <div className="rounded-lg border border-gray-200 bg-white p-3 dark:border-gray-800 dark:bg-gray-900">
    <p className="text-gray-500 text-sm dark:text-gray-400">
      {props.status !== "active"
        ? "Store credit account is not active."
        : "No store credit available."}
    </p>
  </div>
) : (
  <div className="rounded-lg border border-gray-200 bg-white p-3 dark:border-gray-800 dark:bg-gray-900">
    <div className="flex items-center justify-between">
      <div>
        <p className="text-gray-500 text-xs dark:text-gray-400">Available Credit</p>
        <p className="font-semibold text-gray-900 text-sm dark:text-gray-100">
          {props.formatCurrency(props.balance, props.currency)}
        </p>
      </div>
      <button
        type="button"
        onClick={props.onApply}
        disabled={props.isLoading || props.maxApplicable <= 0}
        className="rounded-md bg-gray-900 px-3 py-1.5 text-sm text-white disabled:opacity-50 dark:bg-gray-100 dark:text-gray-900"
      >
        {props.isLoading
          ? "Applying..."
          : `Apply ${props.formatCurrency(props.maxApplicable, props.currency)}`}
      </button>
    </div>
    {props.error && (
      <p className="mt-2 text-red-600 text-sm dark:text-red-400">{props.error}</p>
    )}
  </div>
)}
