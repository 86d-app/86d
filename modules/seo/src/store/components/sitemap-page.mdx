{props.loading && (
  <div className="space-y-3">
    <div className="h-6 w-32 animate-pulse rounded bg-muted" />
    {Array.from({ length: 6 }).map((_, i) => (
      <div key={i} className="h-5 animate-pulse rounded bg-muted" style={{ width: `${60 + Math.random() * 30}%` }} />
    ))}
  </div>
)}

{props.error && (
  <p className="text-sm text-destructive">{props.error}</p>
)}

{!props.loading && !props.error && props.entries.length === 0 && (
  <p className="text-sm text-muted-foreground">No pages available.</p>
)}

{!props.loading && !props.error && props.entries.length > 0 && (
  <div>
    <h2 className="font-display text-lg font-semibold text-foreground">Sitemap</h2>
    <p className="mt-1 text-sm text-muted-foreground">
      All pages on this site ({props.entries.length} {props.entries.length === 1 ? "page" : "pages"}).
    </p>

    <ul className="mt-4 space-y-1.5" role="list">
      {props.entries.map((entry) => (
        <li key={entry.path} className="flex items-baseline justify-between gap-4">
          <a
            href={entry.path}
            className="text-sm font-medium text-foreground underline-offset-2 hover:underline"
          >
            {entry.label}
          </a>
          {entry.lastModified && (
            <span className="shrink-0 text-xs text-muted-foreground">{entry.lastModified}</span>
          )}
        </li>
      ))}
    </ul>
  </div>
)}
