<div className="rounded-lg border border-border bg-muted/20 p-5">
  <h3 className="font-display text-base font-semibold text-foreground mb-4">Order summary</h3>

  {/* Line items */}
  {props.lineItems.length > 0 && (
    <ul className="divide-y divide-border/40 mb-4">
      {props.lineItems.map((item) => (
        <li key={`${item.productId}-${item.variantId ?? ""}`} className="flex items-center justify-between py-2.5">
          <div className="min-w-0 pr-3">
            <p className="text-foreground text-sm truncate">{item.name}</p>
            <p className="text-muted-foreground text-xs">
              {item.formattedPrice} &times; {item.quantity}
            </p>
          </div>
          <p className="flex-shrink-0 text-foreground text-sm tabular-nums">{item.formattedTotal}</p>
        </li>
      ))}
    </ul>
  )}

  {/* Promo code */}
  {props.discountCode ? (
    <div className="flex items-center justify-between rounded-md border border-green-500/30 bg-green-500/5 px-3 py-2 mb-3">
      <div>
        <p className="text-foreground text-xs font-medium">
          Promo: <span className="font-mono">{props.discountCode}</span>
        </p>
        {props.discountAmount && (
          <p className="text-green-600 dark:text-green-400 text-xs">−{props.discountAmount}</p>
        )}
      </div>
      <button
        type="button"
        onClick={props.onRemovePromo}
        className="text-muted-foreground text-xs hover:text-destructive transition-colors"
      >
        Remove
      </button>
    </div>
  ) : (
    <form onSubmit={props.onApplyPromo} className="flex gap-2 mb-3">
      <input
        type="text"
        value={props.promoCode}
        onChange={(e) => props.onPromoCodeChange(e.target.value)}
        placeholder="Promo code"
        disabled={props.promoLoading}
        className="flex-1 rounded-md border border-border bg-background px-3 py-2 text-foreground text-sm placeholder:text-muted-foreground/50 focus:border-foreground focus:outline-none focus:ring-1 focus:ring-foreground disabled:opacity-50"
      />
      <button
        type="submit"
        disabled={props.promoLoading || !props.promoCode.trim()}
        className="rounded-md border border-border px-3 py-2 text-foreground text-sm transition-colors hover:bg-muted disabled:opacity-50"
      >
        Apply
      </button>
    </form>
  )}
  {props.promoError && (
    <p className="text-destructive text-xs mb-3">{props.promoError}</p>
  )}

  {/* Gift card */}
  {props.giftCardCode ? (
    <div className="flex items-center justify-between rounded-md border border-green-500/30 bg-green-500/5 px-3 py-2 mb-4">
      <div>
        <p className="text-foreground text-xs font-medium">
          Gift card: <span className="font-mono">{props.giftCardCode}</span>
        </p>
        {props.giftCardAmount && (
          <p className="text-green-600 dark:text-green-400 text-xs">−{props.giftCardAmount}</p>
        )}
      </div>
      <button
        type="button"
        onClick={props.onRemoveGiftCard}
        className="text-muted-foreground text-xs hover:text-destructive transition-colors"
      >
        Remove
      </button>
    </div>
  ) : (
    <form onSubmit={props.onApplyGiftCard} className="flex gap-2 mb-4">
      <input
        type="text"
        value={props.giftCode}
        onChange={(e) => props.onGiftCodeChange(e.target.value)}
        placeholder="Gift card"
        disabled={props.giftLoading}
        className="flex-1 rounded-md border border-border bg-background px-3 py-2 text-foreground text-sm placeholder:text-muted-foreground/50 focus:border-foreground focus:outline-none focus:ring-1 focus:ring-foreground disabled:opacity-50"
      />
      <button
        type="submit"
        disabled={props.giftLoading || !props.giftCode.trim()}
        className="rounded-md border border-border px-3 py-2 text-foreground text-sm transition-colors hover:bg-muted disabled:opacity-50"
      >
        Apply
      </button>
    </form>
  )}
  {props.giftError && (
    <p className="text-destructive text-xs mb-4">{props.giftError}</p>
  )}

  {/* Totals */}
  <div className="space-y-1.5 border-t border-border pt-3 text-sm">
    <div className="flex justify-between">
      <span className="text-muted-foreground">Subtotal</span>
      <span className="text-foreground tabular-nums">{props.subtotal}</span>
    </div>
    {props.shipping && (
      <div className="flex justify-between">
        <span className="text-muted-foreground">Shipping</span>
        <span className="text-foreground tabular-nums">{props.shipping}</span>
      </div>
    )}
    {props.tax && (
      <div className="flex justify-between">
        <span className="text-muted-foreground">Tax</span>
        <span className="text-foreground tabular-nums">{props.tax}</span>
      </div>
    )}
    {props.discountAmount && (
      <div className="flex justify-between">
        <span className="text-muted-foreground">Discount</span>
        <span className="text-green-600 dark:text-green-400 tabular-nums">−{props.discountAmount}</span>
      </div>
    )}
    {props.giftCardAmount && (
      <div className="flex justify-between">
        <span className="text-muted-foreground">Gift card</span>
        <span className="text-green-600 dark:text-green-400 tabular-nums">−{props.giftCardAmount}</span>
      </div>
    )}
    <div className="flex justify-between border-t border-border pt-2 font-medium">
      <span className="text-foreground">Total</span>
      <span className="font-display text-foreground tabular-nums">{props.total}</span>
    </div>
  </div>
</div>
