{props.orderPlaced ? (
  <div className="space-y-5 text-center py-10">
    <div className="mx-auto flex h-14 w-14 items-center justify-center rounded-full bg-green-500/10">
      <svg xmlns="http://www.w3.org/2000/svg" width="28" height="28" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className="text-green-600 dark:text-green-400" aria-hidden="true">
        <path d="M20 6 9 17l-5-5" />
      </svg>
    </div>
    <div>
      <h2 className="font-display text-xl font-semibold text-foreground">Order placed!</h2>
      {props.orderId && (
        <p className="mt-1 text-muted-foreground text-sm">
          Order number: <span className="font-mono text-foreground">{props.orderId}</span>
        </p>
      )}
      <p className="mt-3 text-muted-foreground text-sm">
        You'll receive a confirmation email shortly.
      </p>
    </div>
    <a
      href="/"
      className="inline-block rounded-md bg-foreground px-6 py-2.5 font-medium text-background text-sm transition-opacity hover:opacity-85"
    >
      Continue shopping
    </a>
  </div>
) : (
  <div className="space-y-5">
    <div>
      <h2 className="font-display text-lg font-semibold text-foreground">Review your order</h2>
    </div>

    {props.session && (
      <>
        {/* Contact */}
        <div className="rounded-md border border-border p-4">
          <p className="text-muted-foreground text-xs font-medium uppercase tracking-wider mb-1">Contact</p>
          <p className="text-foreground text-sm">{props.session.guestEmail ?? "—"}</p>
        </div>

        {/* Shipping address */}
        {props.shippingAddress && (
          <div className="rounded-md border border-border p-4">
            <p className="text-muted-foreground text-xs font-medium uppercase tracking-wider mb-1">Ship to</p>
            <p className="text-foreground text-sm">{props.shippingAddress}</p>
          </div>
        )}

        {/* Line items */}
        {props.formattedItems.length > 0 && (
          <div className="rounded-md border border-border divide-y divide-border">
            {props.formattedItems.map((item) => (
              <div key={`${item.productId}-${item.variantId ?? ""}`} className="flex items-center justify-between px-4 py-3">
                <div className="min-w-0">
                  <p className="text-foreground text-sm truncate">{item.name}</p>
                  <p className="text-muted-foreground text-xs">
                    {item.formattedPrice} &times; {item.quantity}
                  </p>
                </div>
                <p className="flex-shrink-0 text-foreground text-sm tabular-nums ml-4">
                  {item.formattedTotal}
                </p>
              </div>
            ))}
          </div>
        )}

        {/* Totals */}
        <div className="space-y-1.5 text-sm">
          <div className="flex justify-between">
            <span className="text-muted-foreground">Subtotal</span>
            <span className="text-foreground tabular-nums">{props.formatPrice(props.session.subtotal)}</span>
          </div>
          {props.session.shippingAmount > 0 && (
            <div className="flex justify-between">
              <span className="text-muted-foreground">Shipping</span>
              <span className="text-foreground tabular-nums">{props.formatPrice(props.session.shippingAmount)}</span>
            </div>
          )}
          {props.session.taxAmount > 0 && (
            <div className="flex justify-between">
              <span className="text-muted-foreground">Tax</span>
              <span className="text-foreground tabular-nums">{props.formatPrice(props.session.taxAmount)}</span>
            </div>
          )}
          {props.session.discountAmount > 0 && (
            <div className="flex justify-between">
              <span className="text-muted-foreground">
                Discount{props.session.discountCode ? ` (${props.session.discountCode})` : ""}
              </span>
              <span className="text-green-600 dark:text-green-400 tabular-nums">
                −{props.formatPrice(props.session.discountAmount)}
              </span>
            </div>
          )}
          {props.session.giftCardAmount > 0 && (
            <div className="flex justify-between">
              <span className="text-muted-foreground">Gift card</span>
              <span className="text-green-600 dark:text-green-400 tabular-nums">
                −{props.formatPrice(props.session.giftCardAmount)}
              </span>
            </div>
          )}
          <div className="flex justify-between border-t border-border pt-2 font-medium">
            <span className="text-foreground">Total</span>
            <span className="font-display text-foreground tabular-nums">{props.formatPrice(props.session.total)}</span>
          </div>
        </div>
      </>
    )}

    {props.error && (
      <p className="text-destructive text-sm">{props.error}</p>
    )}

    <div className="flex items-center gap-3">
      <button
        type="button"
        onClick={props.onBack}
        disabled={props.loading}
        className="rounded-md border border-border px-4 py-2.5 text-foreground text-sm transition-colors hover:bg-muted disabled:opacity-50"
      >
        Back
      </button>
      <button
        type="button"
        onClick={props.onPlaceOrder}
        disabled={props.loading}
        className="flex-1 rounded-md bg-foreground py-2.5 text-center font-medium text-background text-sm transition-opacity hover:opacity-85 disabled:opacity-50"
      >
        {props.loading ? "Placing order…" : "Place order"}
      </button>
    </div>
  </div>
)}
