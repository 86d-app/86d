  <nav className="grid grid-cols-2 gap-8 sm:grid-cols-3 lg:grid-cols-4" aria-label="Footer">
    {Array.from({ length: 4 }).map((_, i) => (
      <div key={`footer-skeleton-${i}`} className="space-y-3">

        <div className="h-3 w-28 animate-pulse rounded bg-muted" />
      </div>
    ))}
  </nav>

  <nav

    aria-label={props.menuName || "Footer navigation"}
  >
    {props.columns.map((col) => (

            <h3 className="font-semibold text-foreground text-sm tracking-wide">

            <ul className="mt-3 space-y-2">

                  >
                    {child.label}
                  </a>
                </li>
              ))}
            </ul>
          </>
        ) : (
          <a
            href={col.href}
            target={col.openInNewTab ? "_blank" : undefined}

          </a>
        )}
      </div>
