{props.loading ? (
  <div className="py-12 text-center text-muted-foreground">Loading registry…</div>
) : !props.registry ? (
  <div className="py-12 text-center text-muted-foreground">Registry not found.</div>
) : (
  <div className="space-y-6">
    <div>
      <h1 className="text-3xl font-bold">{props.registry.title}</h1>
      <p className="text-muted-foreground">
        by {props.registry.customerName}
      </p>
      {props.registry.eventDate && (
        <p className="text-sm text-muted-foreground">
          Event date: {new Date(props.registry.eventDate).toLocaleDateString()}
        </p>
      )}
      {props.registry.description && (
        <p className="mt-3">{props.registry.description}</p>
      )}
    </div>

    <div className="text-sm text-muted-foreground">
      {props.registry.purchasedCount} of {props.registry.itemCount} items purchased
    </div>

    {props.items.length === 0 ? (
      <p className="py-8 text-center text-muted-foreground">No items on this registry yet.</p>
    ) : (
      <div className="grid gap-4 sm:grid-cols-2 lg:grid-cols-3">
        {props.items.map((item) => {
          const remaining = item.quantityDesired - item.quantityReceived;
          const fulfilled = remaining <= 0;
          return (
            <div
              key={item.id}
              className={`rounded-lg border p-4 ${fulfilled ? "opacity-60" : ""}`}
            >
              <div className="mb-1 text-xs font-medium uppercase tracking-wider text-muted-foreground">
                {props.priorityLabels[item.priority] || item.priority}
              </div>
              <h3 className="font-semibold">{item.productName}</h3>
              {item.variantName && (
                <p className="text-sm text-muted-foreground">{item.variantName}</p>
              )}
              <p className="mt-2 text-lg font-semibold">
                ${(item.priceInCents / 100).toFixed(2)}
              </p>
              {item.note && (
                <p className="mt-1 text-sm italic text-muted-foreground">"{item.note}"</p>
              )}
              <div className="mt-3 text-sm">
                {fulfilled ? (
                  <span className="font-medium text-green-600">Fulfilled</span>
                ) : (
                  <span>
                    <span className="font-medium">{remaining}</span> still needed
                  </span>
                )}
              </div>
            </div>
          );
        })}
      </div>
    )}

    {props.registry.thankYouMessage && (
      <div className="rounded-lg border bg-muted/30 p-4">
        <p className="text-sm font-medium">A note from {props.registry.customerName}:</p>
        <p className="mt-1 italic">{props.registry.thankYouMessage}</p>
      </div>
    )}
  </div>
)}
