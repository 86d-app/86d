# Pickup Locations

Manage physical locations where customers can pick up their orders.

<div className="space-y-4">
  {props.summary && (
    <div className="grid grid-cols-4 gap-4">
      <div className="rounded-lg border p-4">
        <div className="text-sm text-muted-foreground">Locations</div>
        <div className="text-2xl font-bold">{props.summary.totalLocations}</div>
      </div>
      <div className="rounded-lg border p-4">
        <div className="text-sm text-muted-foreground">Active</div>
        <div className="text-2xl font-bold">{props.summary.activeLocations}</div>
      </div>
      <div className="rounded-lg border p-4">
        <div className="text-sm text-muted-foreground">Scheduled</div>
        <div className="text-2xl font-bold">{props.summary.scheduledPickups}</div>
      </div>
      <div className="rounded-lg border p-4">
        <div className="text-sm text-muted-foreground">Ready</div>
        <div className="text-2xl font-bold">{props.summary.readyPickups}</div>
      </div>
    </div>
  )}

  <div className="flex items-center gap-2">
    <select
      className="rounded border px-3 py-1.5 text-sm"
      value={props.activeFilter}
      onChange={(e) => props.onActiveChange(e.target.value)}
    >
      <option value="">All</option>
      <option value="true">Active</option>
      <option value="false">Inactive</option>
    </select>
  </div>

  {props.loading ? (
    <div className="py-12 text-center text-muted-foreground">Loading pickup locations…</div>
  ) : props.locations.length === 0 ? (
    <div className="py-12 text-center text-muted-foreground">No pickup locations found.</div>
  ) : (
    <div className="overflow-x-auto rounded-lg border">
      <table className="w-full text-sm">
        <thead className="border-b bg-muted/50">
          <tr>
            <th className="px-4 py-3 text-left font-medium">Name</th>
            <th className="px-4 py-3 text-left font-medium">Address</th>
            <th className="px-4 py-3 text-left font-medium">Prep Time</th>
            <th className="px-4 py-3 text-left font-medium">Status</th>
          </tr>
        </thead>
        <tbody>
          {props.locations.map((loc) => (
            <tr key={loc.id} className="border-b last:border-0 hover:bg-muted/30">
              <td className="px-4 py-3 font-medium">{loc.name}</td>
              <td className="px-4 py-3">{loc.address}, {loc.city}, {loc.state} {loc.postalCode}</td>
              <td className="px-4 py-3">{loc.preparationMinutes} min</td>
              <td className="px-4 py-3">
                <span className={`inline-flex rounded-full border px-2 py-0.5 text-xs font-medium ${loc.active ? "text-green-700 bg-green-50 border-green-200" : "text-gray-700 bg-gray-50 border-gray-200"}`}>
                  {loc.active ? "Active" : "Inactive"}
                </span>
              </td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  )}
</div>
