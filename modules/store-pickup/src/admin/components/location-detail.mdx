# Location Detail

View and manage a pickup location's time windows.

<div className="space-y-4">
  {props.loading ? (
    <div className="py-12 text-center text-muted-foreground">Loading location…</div>
  ) : !props.location ? (
    <div className="py-12 text-center text-muted-foreground">Location not found.</div>
  ) : (
    <>
      <div className="rounded-lg border p-4 space-y-1">
        <h2 className="text-lg font-semibold">{props.location.name}</h2>
        <p className="text-sm text-muted-foreground">
          {props.location.address}, {props.location.city}, {props.location.state} {props.location.postalCode}
        </p>
        {props.location.phone && (
          <p className="text-sm text-muted-foreground">{props.location.phone}</p>
        )}
        {props.location.email && (
          <p className="text-sm text-muted-foreground">{props.location.email}</p>
        )}
        <div className="flex gap-4 pt-2">
          <span className="text-sm">Prep time: <strong>{props.location.preparationMinutes} min</strong></span>
          <span className={`inline-flex rounded-full border px-2 py-0.5 text-xs font-medium ${props.location.active ? "text-green-700 bg-green-50 border-green-200" : "text-gray-700 bg-gray-50 border-gray-200"}`}>
            {props.location.active ? "Active" : "Inactive"}
          </span>
        </div>
      </div>

      <h3 className="text-base font-semibold">Pickup Windows</h3>

      {props.windows.length === 0 ? (
        <div className="py-8 text-center text-muted-foreground">No pickup windows configured.</div>
      ) : (
        <div className="overflow-x-auto rounded-lg border">
          <table className="w-full text-sm">
            <thead className="border-b bg-muted/50">
              <tr>
                <th className="px-4 py-3 text-left font-medium">Day</th>
                <th className="px-4 py-3 text-left font-medium">Window</th>
                <th className="px-4 py-3 text-left font-medium">Capacity</th>
                <th className="px-4 py-3 text-left font-medium">Status</th>
              </tr>
            </thead>
            <tbody>
              {props.windows.map((w) => (
                <tr key={w.id} className="border-b last:border-0 hover:bg-muted/30">
                  <td className="px-4 py-3">{props.dayNames[w.dayOfWeek]}</td>
                  <td className="px-4 py-3">{w.startTime} – {w.endTime}</td>
                  <td className="px-4 py-3">{w.capacity}</td>
                  <td className="px-4 py-3">
                    <span className={`inline-flex rounded-full border px-2 py-0.5 text-xs font-medium ${w.active ? "text-green-700 bg-green-50 border-green-200" : "text-gray-700 bg-gray-50 border-gray-200"}`}>
                      {w.active ? "Active" : "Inactive"}
                    </span>
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      )}
    </>
  )}
</div>
