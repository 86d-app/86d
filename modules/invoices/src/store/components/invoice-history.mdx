## My Invoices

<div className="space-y-4">
  {props.loading ? (
    <div className="py-8 text-center text-muted-foreground">Loading invoices…</div>
  ) : props.invoices.length === 0 ? (
    <div className="py-8 text-center text-muted-foreground">You have no invoices yet.</div>
  ) : (
    <div className="space-y-3">
      {props.invoices.map((inv) => (
        <div key={inv.id} className="flex items-center justify-between rounded-lg border p-4">
          <div>
            <div className="font-medium">{inv.invoiceNumber}</div>
            <div className="text-sm text-muted-foreground">{new Date(inv.createdAt).toLocaleDateString()}</div>
          </div>
          <div className="text-right">
            <div className="font-medium">${(inv.total / 100).toFixed(2)}</div>
            <span className={`inline-flex rounded-full border px-2 py-0.5 text-xs font-medium ${
              inv.status === "paid" ? "text-green-700 bg-green-50 border-green-200" :
              inv.status === "overdue" ? "text-red-700 bg-red-50 border-red-200" :
              inv.status === "partially_paid" ? "text-yellow-700 bg-yellow-50 border-yellow-200" :
              "text-gray-700 bg-gray-50 border-gray-200"
            }`}>
              {inv.status.replace("_", " ")}
            </span>
          </div>
        </div>
      ))}
    </div>
  )}

  {props.pages > 1 && (
    <div className="flex justify-center gap-2">
      <button
        className="rounded border px-3 py-1 text-sm disabled:opacity-50"
        disabled={props.page <= 1}
        onClick={() => props.onPageChange(props.page - 1)}
      >
        Previous
      </button>
      <button
        className="rounded border px-3 py-1 text-sm disabled:opacity-50"
        disabled={props.page >= props.pages}
        onClick={() => props.onPageChange(props.page + 1)}
      >
        Next
      </button>
    </div>
  )}
</div>
