<div className="rounded-lg border border-border p-4">
  {props.isCancelled ? (
    <div className="flex items-center gap-2 py-2">
      <svg xmlns="http://www.w3.org/2000/svg" width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={props.cancelledColor} aria-hidden="true">
        <circle cx="12" cy="12" r="10" />
        <path d="m15 9-6 6" />
        <path d="m9 9 6 6" />
      </svg>
      <span className={`text-sm font-medium ${props.cancelledColor}`}>{props.cancelledLabel}</span>
    </div>
  ) : (
    <div className="flex items-center justify-between">
      {props.steps.map((step, i) => (
        <div key={step.key} className="flex flex-1 items-center">
          <div className="flex flex-col items-center gap-1">
            <div
              className={`flex h-7 w-7 items-center justify-center rounded-full border-2 text-xs font-semibold transition-colors ${
                step.isCompleted
                  ? "border-green-600 bg-green-600 text-white dark:border-green-500 dark:bg-green-500"
                  : "border-border bg-background text-muted-foreground"
              }`}
            >
              {step.isCompleted ? (
                <svg xmlns="http://www.w3.org/2000/svg" width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="3" strokeLinecap="round" strokeLinejoin="round" aria-hidden="true">
                  <path d="M20 6 9 17l-5-5" />
                </svg>
              ) : (
                <span>{i + 1}</span>
              )}
            </div>
            <span className={`text-xs font-medium ${step.isCurrent ? "text-foreground" : "text-muted-foreground"}`}>
              {step.label}
            </span>
            {step.timestamp && (
              <span className="text-[10px] text-muted-foreground">{step.timestamp}</span>
            )}
          </div>
          {i < props.steps.length - 1 && (
            <div
              className={`mx-2 h-0.5 flex-1 rounded transition-colors ${
                step.isCompleted && props.steps[i + 1]?.isCompleted
                  ? "bg-green-600 dark:bg-green-500"
                  : "bg-border"
              }`}
            />
          )}
        </div>
      ))}
    </div>
  )}
</div>
