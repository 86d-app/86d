{props.result && props.appliedCode ? (
  <div className="flex items-center justify-between gap-3 rounded-lg border border-emerald-200 bg-emerald-50 px-4 py-3 dark:border-emerald-800 dark:bg-emerald-950/30">
    <div className="min-w-0">
      <div className="flex items-center gap-2">
        <span className="font-medium font-mono text-emerald-800 text-sm dark:text-emerald-200">
          {props.appliedCode}
        </span>
        <span className="rounded-full bg-emerald-100 px-2 py-0.5 text-emerald-700 text-xs dark:bg-emerald-900 dark:text-emerald-300">
          Applied
        </span>
      </div>
      <p className="mt-0.5 text-emerald-700 text-xs dark:text-emerald-300">
        {props.result.freeShipping
          ? "Free shipping"
          : `−${props.formatCents(props.result.discountAmount)} off`}
      </p>
    </div>
    <button
      type="button"
      onClick={props.onRemove}
      className="shrink-0 text-emerald-600 text-sm underline-offset-4 hover:underline dark:text-emerald-400"
    >
      Remove
    </button>
  </div>
) : props.compact ? (
  <form onSubmit={props.onSubmit} className="space-y-1.5">
    <div className="flex gap-2">
      <input
        type="text"
        value={props.code}
        onChange={(e) => props.onCodeChange(e.target.value)}
        placeholder="Promo code"
        maxLength={100}
        className="min-w-0 flex-1 rounded-lg border border-input bg-background px-3 py-2 font-mono text-sm uppercase outline-none ring-offset-background placeholder:font-sans placeholder:normal-case focus:ring-2 focus:ring-ring focus:ring-offset-1"
      />
      <button
        type="submit"
        disabled={props.isLoading || !props.code.trim()}
        className="shrink-0 rounded-lg border border-border bg-background px-4 py-2 font-medium text-foreground text-sm transition-colors hover:bg-muted disabled:opacity-60"
      >
        {props.isLoading ? "…" : "Apply"}
      </button>
    </div>
    {props.error && (
      <p className="text-destructive text-xs" role="alert">
        {props.error}
      </p>
    )}
  </form>
) : (
  <form
    onSubmit={props.onSubmit}
    className="space-y-3 rounded-xl border border-border bg-muted/30 p-4"
  >
    <p className="font-medium text-foreground text-sm">Have a promo code?</p>
    <div className="flex gap-2">
      <input
        type="text"
        value={props.code}
        onChange={(e) => props.onCodeChange(e.target.value)}
        placeholder="Enter code"
        maxLength={100}
        className="min-w-0 flex-1 rounded-lg border border-input bg-background px-3 py-2 font-mono text-sm uppercase outline-none ring-offset-background placeholder:font-sans placeholder:normal-case focus:ring-2 focus:ring-ring focus:ring-offset-1"
      />
      <button
        type="submit"
        disabled={props.isLoading || !props.code.trim()}
        className="shrink-0 rounded-lg bg-primary px-4 py-2 font-medium text-primary-foreground text-sm transition-opacity disabled:opacity-60"
      >
        {props.isLoading ? "Validating…" : "Apply"}
      </button>
    </div>
    {props.error && (
      <p className="text-destructive text-sm" role="alert">
        {props.error}
      </p>
    )}
  </form>
)}
