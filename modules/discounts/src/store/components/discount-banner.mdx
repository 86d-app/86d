{props.items.length === 1 ? (
  <div className="rounded-xl border border-amber-200 bg-gradient-to-r from-amber-50 to-orange-50 px-5 py-4 dark:border-amber-800 dark:from-amber-950/30 dark:to-orange-950/30">
    <div className="flex items-center gap-3">
      <span className="rounded-lg bg-amber-100 px-2.5 py-1 font-bold text-amber-800 text-sm dark:bg-amber-900 dark:text-amber-200">
        {props.items[0].badge}
      </span>
      <div className="min-w-0 flex-1">
        <p className="font-semibold text-foreground text-sm">{props.items[0].name}</p>
        {props.items[0].description && (
          <p className="text-muted-foreground text-xs">{props.items[0].description}</p>
        )}
      </div>
      <div className="shrink-0 text-right text-xs">
        {props.items[0].hasMinimum && (
          <p className="text-muted-foreground">Min. {props.items[0].minimumFormatted}</p>
        )}
        {props.items[0].hasExpiry && (
          <p className="font-medium text-amber-700 dark:text-amber-300">{props.items[0].expiryFormatted}</p>
        )}
      </div>
    </div>
  </div>
) : (
  <div className="space-y-2">
    {props.items.map((item) => (
      <div
        key={item.id}
        className="flex items-center gap-3 rounded-lg border border-border bg-muted/30 px-4 py-3"
      >
        <span className="rounded-md bg-amber-100 px-2 py-0.5 font-bold text-amber-800 text-xs dark:bg-amber-900 dark:text-amber-200">
          {item.badge}
        </span>
        <div className="min-w-0 flex-1">
          <p className="font-medium text-foreground text-sm">{item.name}</p>
          {item.description && (
            <p className="truncate text-muted-foreground text-xs">{item.description}</p>
          )}
        </div>
        <div className="shrink-0 text-right text-xs text-muted-foreground">
          {item.hasMinimum && <p>Min. {item.minimumFormatted}</p>}
          {item.hasExpiry && (
            <p className="font-medium text-amber-700 dark:text-amber-300">{item.expiryFormatted}</p>
          )}
        </div>
      </div>
    ))}
  </div>
)}
