<div className="space-y-6">
  <div className="flex items-center justify-between">
    <div>
      <h2 className="font-semibold text-2xl">Reward Rules</h2>
      <p className="text-muted-foreground text-sm">Configure rewards for referrers and referees</p>
    </div>
    <button
      type="button"
      onClick={props.onToggleForm}
      className="rounded-md bg-primary px-4 py-2 font-medium text-primary-foreground text-sm hover:bg-primary/90"
    >
      {props.showForm ? "Cancel" : "New Rule"}
    </button>
  </div>

  {props.showForm && (
    <form onSubmit={props.onSubmit} className="space-y-4 rounded-lg border bg-card p-4">
      <div>
        <label className="mb-1 block font-medium text-sm">Rule Name</label>
        <input
          type="text"
          value={props.name}
          onChange={(e) => props.onNameChange(e.target.value)}
          className="w-full rounded-md border bg-background px-3 py-2 text-sm"
          placeholder="e.g., Welcome Referral"
          required
        />
      </div>

      <div className="grid grid-cols-2 gap-4">
        <div>
          <label className="mb-1 block font-medium text-sm">Referrer Reward</label>
          <select
            value={props.referrerType}
            onChange={(e) => props.onReferrerTypeChange(e.target.value)}
            className="mb-2 w-full rounded-md border bg-background px-3 py-2 text-sm"
          >
            <option value="percentage_discount">% Discount</option>
            <option value="fixed_discount">Fixed Discount</option>
            <option value="store_credit">Store Credit</option>
          </select>
          <input
            type="number"
            value={props.referrerValue}
            onChange={(e) => props.onReferrerValueChange(e.target.value)}
            className="w-full rounded-md border bg-background px-3 py-2 text-sm"
            min="0"
            step="0.01"
            required
          />
        </div>
        <div>
          <label className="mb-1 block font-medium text-sm">Referee Reward</label>
          <select
            value={props.refereeType}
            onChange={(e) => props.onRefereeTypeChange(e.target.value)}
            className="mb-2 w-full rounded-md border bg-background px-3 py-2 text-sm"
          >
            <option value="percentage_discount">% Discount</option>
            <option value="fixed_discount">Fixed Discount</option>
            <option value="store_credit">Store Credit</option>
          </select>
          <input
            type="number"
            value={props.refereeValue}
            onChange={(e) => props.onRefereeValueChange(e.target.value)}
            className="w-full rounded-md border bg-background px-3 py-2 text-sm"
            min="0"
            step="0.01"
            required
          />
        </div>
      </div>

      <div>
        <label className="mb-1 block font-medium text-sm">Minimum Order Amount</label>
        <input
          type="number"
          value={props.minOrderAmount}
          onChange={(e) => props.onMinOrderAmountChange(e.target.value)}
          className="w-full rounded-md border bg-background px-3 py-2 text-sm"
          min="0"
          step="0.01"
        />
      </div>

      <button
        type="submit"
        disabled={props.isCreating}
        className="rounded-md bg-primary px-4 py-2 font-medium text-primary-foreground text-sm hover:bg-primary/90 disabled:opacity-50"
      >
        {props.isCreating ? "Creating..." : "Create Rule"}
      </button>
    </form>
  )}

  {props.loading ? (
    <p className="py-12 text-center text-muted-foreground">Loading rules...</p>
  ) : props.rules.length === 0 ? (
    <p className="py-12 text-center text-muted-foreground">No reward rules configured</p>
  ) : (
    <div className="space-y-3">
      {props.rules.map((rule) => (
        <div key={rule.id} className="flex items-center justify-between rounded-lg border bg-card p-4">
          <div className="space-y-1">
            <div className="flex items-center gap-2">
              <h3 className="font-medium">{rule.name}</h3>
              <span className={`inline-flex rounded-full px-2 py-0.5 text-xs ${rule.active ? "bg-green-100 text-green-800 dark:bg-green-900/30 dark:text-green-400" : "bg-gray-100 text-gray-800 dark:bg-gray-900/30 dark:text-gray-400"}`}>
                {rule.active ? "Active" : "Inactive"}
              </span>
            </div>
            <p className="text-muted-foreground text-sm">
              Referrer: {props.formatReward(rule.referrerRewardType, rule.referrerRewardValue)} · Referee: {props.formatReward(rule.refereeRewardType, rule.refereeRewardValue)}
              {rule.minOrderAmount > 0 && ` · Min order: $${rule.minOrderAmount.toFixed(2)}`}
            </p>
          </div>
          <div className="flex gap-2">
            <button
              type="button"
              onClick={() => props.onToggleActive(rule.id, !rule.active)}
              className="rounded px-3 py-1.5 text-sm hover:bg-muted"
            >
              {rule.active ? "Disable" : "Enable"}
            </button>
            <button
              type="button"
              onClick={() => props.onDelete(rule.id)}
              className="rounded px-3 py-1.5 text-red-600 text-sm hover:bg-red-50 dark:hover:bg-red-900/20"
            >
              Delete
            </button>
          </div>
        </div>
      ))}
    </div>
  )}
</div>
