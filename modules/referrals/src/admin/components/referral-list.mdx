<div className="space-y-6">
  <div className="flex items-center justify-between">
    <div>
      <h2 className="font-semibold text-2xl">Referrals</h2>
      <p className="text-muted-foreground text-sm">Manage customer referrals and track conversions</p>
    </div>
  </div>

  {props.stats && (
    <div className="grid grid-cols-2 gap-4 md:grid-cols-4">
      <div className="rounded-lg border bg-card p-4">
        <p className="font-medium text-2xl">{props.stats.totalCodes}</p>
        <p className="text-muted-foreground text-sm">Referral Codes</p>
      </div>
      <div className="rounded-lg border bg-card p-4">
        <p className="font-medium text-2xl">{props.stats.totalReferrals}</p>
        <p className="text-muted-foreground text-sm">Total Referrals</p>
      </div>
      <div className="rounded-lg border bg-card p-4">
        <p className="font-medium text-2xl">{props.stats.completedReferrals}</p>
        <p className="text-muted-foreground text-sm">Completed</p>
      </div>
      <div className="rounded-lg border bg-card p-4">
        <p className="font-medium text-2xl">{(props.stats.conversionRate * 100).toFixed(1)}%</p>
        <p className="text-muted-foreground text-sm">Conversion Rate</p>
      </div>
    </div>
  )}

  <div className="flex items-center gap-2">
    <select
      className="rounded-md border bg-background px-3 py-2 text-sm"
      value={props.statusFilter}
      onChange={(e) => props.onStatusFilterChange(e.target.value)}
    >
      <option value="">All statuses</option>
      <option value="pending">Pending</option>
      <option value="completed">Completed</option>
      <option value="expired">Expired</option>
      <option value="revoked">Revoked</option>
    </select>
  </div>

  {props.loading ? (
    <p className="py-12 text-center text-muted-foreground">Loading referrals...</p>
  ) : props.referrals.length === 0 ? (
    <p className="py-12 text-center text-muted-foreground">No referrals found</p>
  ) : (
    <div className="overflow-x-auto rounded-lg border">
      <table className="w-full text-sm">
        <thead className="border-b bg-muted/50">
          <tr>
            <th className="px-4 py-3 text-left font-medium">Referrer</th>
            <th className="px-4 py-3 text-left font-medium">Referee</th>
            <th className="px-4 py-3 text-left font-medium">Status</th>
            <th className="px-4 py-3 text-left font-medium">Rewards</th>
            <th className="px-4 py-3 text-left font-medium">Created</th>
            <th className="px-4 py-3 text-left font-medium">Actions</th>
          </tr>
        </thead>
        <tbody className="divide-y">
          {props.referrals.map((r) => (
            <tr key={r.id} className="hover:bg-muted/30">
              <td className="px-4 py-3 font-mono text-xs">{r.referrerCustomerId.slice(0, 8)}...</td>
              <td className="px-4 py-3 font-mono text-xs">{r.refereeCustomerId.slice(0, 8)}...</td>
              <td className="px-4 py-3">
                <span className={`inline-flex rounded-full px-2 py-0.5 font-medium text-xs ${props.statusColors[r.status] ?? ""}`}>
                  {r.status}
                </span>
              </td>
              <td className="px-4 py-3 text-xs">
                <span className={r.referrerRewarded ? "text-green-600" : "text-muted-foreground"}>
                  Referrer: {r.referrerRewarded ? "Yes" : "No"}
                </span>
                {" / "}
                <span className={r.refereeRewarded ? "text-green-600" : "text-muted-foreground"}>
                  Referee: {r.refereeRewarded ? "Yes" : "No"}
                </span>
              </td>
              <td className="px-4 py-3 text-muted-foreground text-xs">{props.timeAgo(r.createdAt)}</td>
              <td className="px-4 py-3">
                {r.status === "pending" && (
                  <div className="flex gap-1">
                    <button
                      type="button"
                      onClick={() => props.onComplete(r.id)}
                      className="rounded px-2 py-1 text-xs text-green-600 hover:bg-green-50 dark:hover:bg-green-900/20"
                    >
                      Complete
                    </button>
                    <button
                      type="button"
                      onClick={() => props.onRevoke(r.id)}
                      className="rounded px-2 py-1 text-xs text-red-600 hover:bg-red-50 dark:hover:bg-red-900/20"
                    >
                      Revoke
                    </button>
                  </div>
                )}
              </td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  )}

  <div className="flex items-center justify-between">
    <button
      type="button"
      disabled={props.page <= 1}
      onClick={() => props.onPageChange(props.page - 1)}
      className="rounded-md border px-3 py-1.5 text-sm disabled:opacity-50"
    >
      Previous
    </button>
    <span className="text-muted-foreground text-sm">Page {props.page}</span>

      type="button"
      onClick={() => props.onPageChange(props.page + 1)}
