{props.isLoading ? (
  <div className="space-y-3">
    <div className="h-5 w-24 animate-pulse rounded bg-muted/30" />
    <div className="flex gap-2">
      {Array.from({ length: 4 }).map((_, i) => (
        <div key={`skel-${i}`} className="h-10 w-16 animate-pulse rounded-lg border border-border bg-muted/30" />
      ))}
    </div>
  </div>
) : (
  <div className="space-y-4">
    <div className="flex items-center justify-between">
      <p className="font-medium text-foreground text-sm">Add a tip</p>
      {props.tipTotal > 0 && (
        <p className="text-muted-foreground text-sm">
          Tip: <span className="font-medium text-foreground">{props.formatCurrency(props.tipTotal, props.currency)}</span>
        </p>
      )}
    </div>

    {/* Preset buttons */}
    <div className="flex flex-wrap gap-2">
      {props.presets.map((pct) => {
        const presetAmount = Math.round((props.orderTotal * pct) / 100);
        const isSelected = props.currentTip?.percentage === pct && props.currentTip?.type === "preset";
        return (
          <button
            key={pct}
            type="button"
            disabled={props.isMutating}
            onClick={() => props.onPreset(pct)}
            className={`flex flex-col items-center rounded-lg border px-3 py-2 text-center transition-all disabled:cursor-not-allowed disabled:opacity-50 ${
              isSelected
                ? "border-foreground bg-foreground text-background"
                : "border-border bg-card text-foreground hover:border-foreground/30 hover:bg-muted"
            }`}
          >
            <span className="font-semibold text-sm">{pct}%</span>
            <span className={`text-xs ${isSelected ? "text-background/70" : "text-muted-foreground"}`}>
              {props.formatCurrency(presetAmount, props.currency)}
            </span>
          </button>
        );
      })}
      <button
        type="button"
        disabled={props.isMutating}
        onClick={props.onRemove}
        className={`flex items-center rounded-lg border px-3 py-2 text-sm transition-all disabled:cursor-not-allowed disabled:opacity-50 ${
          !props.currentTip
            ? "border-foreground bg-foreground text-background"
            : "border-border bg-card text-muted-foreground hover:border-foreground/30 hover:text-foreground"
        }`}
      >
        No tip
      </button>
    </div>

    {/* Custom amount */}
    {props.allowCustom && (
      <form onSubmit={props.onCustomSubmit} className="flex items-center gap-2">
        <div className="relative flex-1">
          <span className="pointer-events-none absolute inset-y-0 left-3 flex items-center text-muted-foreground text-sm">
            $
          </span>
          <input
            type="number"
            value={props.customAmount}
            onChange={(e) => props.onCustomAmountChange(e.target.value)}
            placeholder="Custom"
            min="0"
            step="0.01"
            className="w-full rounded-lg border border-border bg-background py-2 pl-7 pr-3 text-sm outline-none focus:border-foreground/40 focus:ring-1 focus:ring-foreground/20"
          />
        </div>
        <button
          type="submit"
          disabled={!props.customAmount || props.isMutating}
          className="rounded-lg border border-border bg-card px-3 py-2 text-sm font-medium transition-all hover:border-foreground/30 hover:bg-muted disabled:cursor-not-allowed disabled:opacity-50"
        >
          Apply
        </button>
      </form>
    )}

    {/* Error */}
    {props.error && (
      <p className="text-destructive text-xs">{props.error}</p>
    )}
  </div>
)}
