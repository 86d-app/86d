    <div className="grid grid-cols-1 gap-4 sm:grid-cols-3">
      <div className="rounded-lg border border-border bg-card p-5">
        <p className="font-medium text-muted-foreground text-sm">

        <p className="mt-1 font-semibold text-2xl text-foreground">
          {props.summary.totalWaiting}

      </div>

        <p className="font-medium text-muted-foreground text-sm">

        </p>
        <p className="mt-1 font-semibold text-2xl text-foreground">
          {props.summary.totalNotified}
        </p>
      </div>
      <div className="rounded-lg border border-border bg-card p-5">
        <p className="font-medium text-muted-foreground text-sm">

        </p>

  {props.summary && props.summary.topProducts.length > 0 && (
    <div className="rounded-lg border border-border bg-card">
      <div className="border-border border-b px-5 py-3">
        <h3 className="font-semibold text-foreground text-sm">
          Most Requested Products
        </h3>
      </div>
      <div className="divide-y divide-border">
        {props.summary.topProducts.map((product) => (
          <div
            key={product.productId}
            className="flex items-center justify-between px-5 py-3"
          >
            <span className="text-foreground text-sm">
              {product.productName}
            </span>
            <div className="flex items-center gap-3">
              <span className="rounded-full bg-muted px-2.5 py-0.5 font-medium text-muted-foreground text-xs">

              </span>
              {props.notifyConfirm === product.productId ? (
                <span className="space-x-2">

                    onClick={() => props.onNotify(product.productId)}

                    type="button"
                    onClick={() => props.onNotifyConfirm(null)}

              ) : (

              )}
            </div>
