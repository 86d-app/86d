{/* Order Activity Timeline — pure presentation */}

export default function OrderActivityTemplate({
  orderId,
  entries,
  filter,
  isLoading,
  noteCount,
  eventCount,
  onFilterChange,
  onDelete,
  onNoteAdded,
  AddNoteForm,
  TimelineItem,
}) {
  return (
    <div className="rounded-lg border border-border">
      {/* Header */}
      <div className="flex items-center justify-between border-border border-b px-4 py-3">
        <h3 className="font-semibold text-foreground text-sm">Activity</h3>
        <div className="flex items-center gap-1 rounded-lg bg-muted/50 p-0.5">
          <button
            type="button"
            onClick={() => onFilterChange("all")}
            className={`rounded-md px-2.5 py-1 text-xs transition-colors ${
              filter === "all"
                ? "bg-background font-medium text-foreground shadow-sm"
                : "text-muted-foreground hover:text-foreground"
            }`}
          >
            All
          </button>
          <button
            type="button"
            onClick={() => onFilterChange("notes")}
            className={`rounded-md px-2.5 py-1 text-xs transition-colors ${
              filter === "notes"
                ? "bg-background font-medium text-foreground shadow-sm"
                : "text-muted-foreground hover:text-foreground"
            }`}
          >
            Notes{noteCount > 0 ? ` (${noteCount})` : ""}
          </button>
          <button
            type="button"
            onClick={() => onFilterChange("events")}
            className={`rounded-md px-2.5 py-1 text-xs transition-colors ${
              filter === "events"
                ? "bg-background font-medium text-foreground shadow-sm"
                : "text-muted-foreground hover:text-foreground"
            }`}
          >
            Events{eventCount > 0 ? ` (${eventCount})` : ""}
          </button>
        </div>
      </div>

      {/* Add Note */}
      <div className="border-border border-b px-4 py-3">
        <AddNoteForm orderId={orderId} onAdded={onNoteAdded} />
      </div>

      {/* Timeline */}
      <div className="p-4">
        {isLoading ? (
          <div className="space-y-4">
            {[1, 2, 3].map((i) => (
              <div key={i} className="flex gap-3">
                <div className="size-[19px] animate-pulse rounded-full bg-muted" />
                <div className="flex-1 space-y-2">
                  <div className="h-4 w-1/3 animate-pulse rounded bg-muted" />
                  <div className="h-3 w-2/3 animate-pulse rounded bg-muted" />
                </div>
              </div>
            ))}
          </div>
        ) : entries.length === 0 ? (
          <p className="py-6 text-center text-muted-foreground/60 text-sm italic">
            {filter === "notes"
              ? "No notes yet"
              : filter === "events"
                ? "No events yet"
                : "No activity yet"}
          </p>
        ) : (
          <div>
            {entries.map((entry, idx) => (
              <TimelineItem
                key={entry.id}
                entry={entry}
                isLast={idx === entries.length - 1}
                onDelete={entry.deletable ? onDelete : undefined}
              />
            ))}
          </div>
        )}
      </div>
    </div>
  );
}
