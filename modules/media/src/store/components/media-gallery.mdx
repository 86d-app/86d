{props.loading && (
  <div className="grid grid-cols-2 gap-3 sm:grid-cols-3 md:grid-cols-4">
    {Array.from({ length: 8 }).map((_, i) => (
      <div key={i} className="aspect-square animate-pulse rounded-lg bg-muted" />
    ))}
  </div>
)}

{props.error && (
  <p className="text-sm text-destructive">{props.error}</p>
)}

{!props.loading && !props.error && props.assets.length === 0 && (
  <div className="flex flex-col items-center justify-center rounded-lg border border-dashed border-border py-12">
    <svg className="mb-2 size-8 text-muted-foreground/50" fill="none" viewBox="0 0 24 24" stroke="currentColor" strokeWidth={1.5}>
      <path strokeLinecap="round" strokeLinejoin="round" d="m2.25 15.75 5.159-5.159a2.25 2.25 0 0 1 3.182 0l5.159 5.159m-1.5-1.5 1.409-1.409a2.25 2.25 0 0 1 3.182 0l2.909 2.909M3.75 21h16.5A2.25 2.25 0 0 0 22.5 18.75V5.25A2.25 2.25 0 0 0 20.25 3H3.75A2.25 2.25 0 0 0 1.5 5.25v13.5A2.25 2.25 0 0 0 3.75 21Z" />
    </svg>
    <p className="text-sm text-muted-foreground">No media found.</p>
  </div>
)}

{!props.loading && props.assets.length > 0 && (
  <>
    <div className="grid grid-cols-2 gap-3 sm:grid-cols-3 md:grid-cols-4">
      {props.assets.map((asset) => (
        <button
          key={asset.id}
          type="button"
          onClick={() => props.onSelect(props.selectedId === asset.id ? null : asset.id)}
          className={`group relative overflow-hidden rounded-lg border transition-all ${
            props.selectedId === asset.id
              ? "border-foreground ring-2 ring-foreground/20"
              : "border-border hover:border-foreground/40"
          }`}
        >
          {asset.isImage ? (
            <img
              src={asset.url}
              alt={asset.altText}
              loading="lazy"
              className="aspect-square w-full object-cover"
            />
          ) : asset.isVideo ? (
            <div className="relative aspect-square w-full bg-muted">
              <video src={asset.url} muted preload="metadata" className="size-full object-cover" />
              <div className="absolute inset-0 flex items-center justify-center">
                <div className="flex size-10 items-center justify-center rounded-full bg-foreground/80">
                  <svg className="size-4 text-background" fill="currentColor" viewBox="0 0 24 24">
                    <path d="M8 5v14l11-7z" />
                  </svg>
                </div>
              </div>
            </div>
          ) : (
            <div className="flex aspect-square w-full items-center justify-center bg-muted">
              <span className="text-xs font-medium uppercase text-muted-foreground">{asset.typeLabel}</span>
            </div>
          )}
          <div className="absolute inset-x-0 bottom-0 bg-gradient-to-t from-black/60 to-transparent p-2 opacity-0 transition-opacity group-hover:opacity-100">
            <p className="truncate text-xs font-medium text-white">{asset.name}</p>
            <p className="text-[10px] text-white/70">{asset.formattedSize}</p>
          </div>
        </button>
      ))}
    </div>

    <div className="mt-3 flex items-center justify-between">
      <button
        type="button"
        onClick={props.onPrev}
        disabled={props.page <= 1}
        className="rounded-md border border-border px-3 py-1.5 text-xs font-medium text-foreground transition-opacity hover:opacity-75 disabled:opacity-30"
      >
        Previous
      </button>
      <span className="text-xs text-muted-foreground">Page {props.page}</span>
      <button
        type="button"
        onClick={props.onNext}
        disabled={!props.hasMore}
        className="rounded-md border border-border px-3 py-1.5 text-xs font-medium text-foreground transition-opacity hover:opacity-75 disabled:opacity-30"
      >
        Next
      </button>
    </div>
  </>
)}
