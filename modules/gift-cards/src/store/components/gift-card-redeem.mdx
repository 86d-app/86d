{props.applied ? (
  <div className="rounded-lg border border-green-200 bg-green-50 p-3 dark:border-green-800 dark:bg-green-950">
    <p className="font-medium text-green-700 text-sm dark:text-green-300">
      Gift card applied: {props.formatCurrency(props.applied.amount, props.applied.currency)}
    </p>
    <p className="mt-0.5 text-green-600 text-xs dark:text-green-400">
      Remaining balance:{" "}
      {props.formatCurrency(props.applied.remaining, props.applied.currency)}
    </p>
  </div>
) : (
  <div>
    <form onSubmit={props.onSubmit} className="flex gap-2">
      <input
        type="text"
        value={props.code}
        onChange={(e) => props.onCodeChange(e.target.value.toUpperCase())}
        className="flex-1 rounded-md border border-gray-300 px-3 py-1.5 font-mono text-sm tracking-wider dark:border-gray-700 dark:bg-gray-900 dark:text-gray-100"
        placeholder="GIFT-XXXX-XXXX-XXXX"
        maxLength={19}
      />
      <button
        type="submit"
        disabled={props.isLoading || !props.code.trim()}
        className="rounded-md bg-gray-900 px-3 py-1.5 text-sm text-white disabled:opacity-50 dark:bg-gray-100 dark:text-gray-900"
      >
        {props.isLoading ? "Applying..." : "Apply"}
      </button>
    </form>
    {props.error && (
      <p className="mt-2 text-red-600 text-sm dark:text-red-400">{props.error}</p>
    )}
  </div>
)}
