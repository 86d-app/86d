<div className="mx-auto max-w-md">
  <form onSubmit={props.onSubmit} className="flex gap-2">
    <input
      type="text"
      value={props.code}
      onChange={(e) => props.onCodeChange(e.target.value.toUpperCase())}
      className="flex-1 rounded-md border border-gray-300 px-3 py-2 font-mono text-sm tracking-wider dark:border-gray-700 dark:bg-gray-900 dark:text-gray-100"
      placeholder="GIFT-XXXX-XXXX-XXXX"
      maxLength={19}
    />
    <button
      type="submit"
      disabled={props.loading || !props.code.trim()}
      className="rounded-md bg-gray-900 px-4 py-2 text-sm text-white disabled:opacity-50 dark:bg-gray-100 dark:text-gray-900"
    >
      {props.loading ? "Checking..." : "Check Balance"}
    </button>
  </form>

  {props.error && (
    <div className="mt-3 rounded-lg bg-red-50 p-3 text-red-700 text-sm dark:bg-red-950 dark:text-red-300">
      {props.error}
    </div>
  )}

  {props.result && (
    <div className="mt-3 rounded-lg border border-gray-200 bg-white p-4 text-center dark:border-gray-800 dark:bg-gray-900">
      {props.result.status === "active" ? (
        <>
          <p className="text-gray-500 text-sm dark:text-gray-400">
            Available Balance
          </p>
          <p className="mt-1 font-bold text-3xl text-gray-900 dark:text-gray-100">
            {props.formatCurrency(props.result.balance, props.result.currency)}
          </p>
        </>
      ) : (
        <p className="font-medium text-amber-600 text-sm dark:text-amber-400">
          This gift card is {props.result.status}.
        </p>
      )}
    </div>
  )}
</div>
